\newcommandx{\tgrph}[3][2=2, 3=0]{
\begin{tikzpicture}[transform shape]
    \foreach \x in {1,...,#1}{
        \pgfmathparse{\x*(360/#1)-(360/#1)+90+#3}
        \node[draw,circle,minimum size=0.5cm] (N-\x) at (\pgfmathresult:#2cm) {\x};
    }
    \foreach \x [count=\xi from 1] in {2,...,#1}{
        \foreach \y in {\x,...,#1}{
            \draw (N-\xi) -- (N-\y);
        }
    }
\end{tikzpicture}
}
